```python
# plugins/life_engine/dream/scheduler.py::should_dream

def should_dream(
    self,
    idle_heartbeat_count: int,
    in_sleep_window: bool,
) -> bool:
    """判断是否应该开始做梦。"""
    if self._is_dreaming:
        return False  # 正在做梦

    # 检查梦境间隔（最少90分钟）
    now = time.time()
    if now - self._last_dream_time < self._dream_interval_seconds:
        return False

    # 夜间睡眠窗口
    if in_sleep_window:
        return True  # 夜间触发

    # 白天nap模式（长时间idle）
    if self._nap_enabled and idle_heartbeat_count >= self._idle_trigger_heartbeats:
        return True  # 白天触发（10次空转心跳 ≈ 5分钟）

    return False
```

### 2.2 事件片段采样

**按重要性加权采样**：

```python
# plugins/life_engine/dream/scheduler.py::sample_replay_episodes

def _sample_replay_episodes(
    self,
    event_history: list[Any],
) -> list[list[np.ndarray]]:
    """按重要性加权采样事件片段用于NREM回放。"""
    if not event_history or not self._snn_bridge:
        return []

    # 事件分块（每块20个事件）
    chunk_size = self._nrem_events_per_episode  # 默认20
    chunks: list[list[Any]] = []
    for i in range(0, len(event_history), chunk_size):
        chunks.append(event_history[i : i + chunk_size])

    if not chunks:
        return []

    # 计算每块的重要性得分
    scores: list[float] = []
    for chunk in chunks:
        score = 0.0

        # 事件类型权重
        for event in chunk:
            etype = _event_type_value(event)

            if etype == "message":
                score += 2.0  # 消息最重要

            elif etype == "tool_call":
                score += 1.5  # 工具调用次重要

            elif etype == "tool_result":
                # 成功工具结果更重要
                if _event_value(event, "tool_success", False):
                    score += 1.0
                else:
                    score += 0.5  # 失败结果较不重要

            elif etype == "heartbeat":
                score += 0.3  # 心跳较不重要

        scores.append(max(score, 0.1))  # 最低得分0.1

    # 概率分布（按得分归一化）
    total_score = sum(scores)
    probs = np.array([score / total_score for score in scores])

    # 随机采样（不重复）
    n_samples = min(self._nrem_replay_episodes, len(chunks))  # 默认3个片段
    indices = np.random.choice(len(chunks), size=n_samples, replace=False, p=probs)

    # 提取特征向量
    episodes: list[list[np.ndarray]] = []
    window_seconds = chunk_size * 180.0  # 每块时间窗口（20 × 180秒 = 1小时）

    for idx in sorted(indices):
        chunk = chunks[idx]
        features = self._snn_bridge.extract_features_from_events(
            chunk,
            window_seconds=window_seconds,
        )
        episodes.append([features])

    return episodes
```

### 2.3 SNN回放与巩固

**NREM回放实现**：

```python
# plugins/life_engine/dream/scheduler.py::run_nrem

async def _run_nrem(self, event_history: list[Any]) -> NREMReport:
    """NREM阶段：将历史事件通过SNN回放。"""
    report = NREMReport()

    if not self._snn or not self._snn_bridge:
        return report

    # 记录回放前的突触权重
    report.weight_before = {
        "syn_in_hid": self._snn.syn_in_hid.get_weight_stats(),
        "syn_hid_out": self._snn.syn_hid_out.get_weight_stats(),
    }

    # 采样回放片段
    episodes = self._sample_replay_episodes(event_history)

    # 回放每个片段（5倍速）
    for features_list in episodes:
        if features_list:
            stats = self._snn.replay_episodes(
                features_list,
                speed_multiplier=self._nrem_speed_multiplier,  # 默认5.0
                reward_signal=0.0,  # NREM阶段reward=0（中性回放）
            )
            report.total_steps += stats.get("steps", 0)
            report.episodes_replayed += 1

    # 稳态调节（突触权重归一化）
    if report.episodes_replayed > 0:
        self._snn.homeostatic_scaling(rate=self._nrem_homeostatic_rate)  # 默认0.02
        report.homeostatic_applied = True

    # 记录回放后的突触权重
    report.weight_after = {
        "syn_in_hid": self._snn.syn_in_hid.get_weight_stats(),
        "syn_hid_out": self._snn.syn_hid_out.get_weight_stats(),
    }

    return report
```

**SNN回放详解**：

```python
# plugins/life_engine/snn/core.py::replay_episodes

def replay_episodes(
    self,
    episodes: list[list[np.ndarray]],
    speed_multiplier: float = 5.0,
    reward_signal: float = 0.0,
) -> dict[str, Any]:
    """回放事件片段（通过SNN真实输入步）。"""
    stats = {
        "steps": 0,
        "total_episodes": len(episodes),
    }

    # 时间压缩（5倍速回放）
    # 原本30秒的事件 → 回放时6秒处理完毕
    for episode in episodes:
        for features in episode:
            # SNN真实输入步（带STDP学习）
            output = self.step(features, reward=reward_signal)
            stats["steps"] += 1

    return stats
```

**稳态调节详解**：

```python
# plugins/life_engine/snn/core.py::homeostatic_scaling

def homeostatic_scaling(self, rate: float = 0.02):
    """突触稳态调节：权重归一化到基线范围。"""
    # 输入层到隐藏层突触
    for synapse in [self.syn_in_hid, self.syn_hid_out]:
        # 计算当前权重均值
        mean_weight = np.mean(synapse._weights)

        # 目标均值（基线）
        target_mean = 0.5

        # 向基线移动（每次调节2%）
        adjustment = rate * (target_mean - mean_weight)
        synapse._weights += adjustment

        # 重新约束到 [w_min, w_max]
        synapse._weights = np.clip(synapse._weights, synapse._w_min, synapse._w_max)
```

**稳态调节的意义**：

| 问题 | 无稳态调节 | 有稳态调节 |
|------|----------|----------|
| **权重爆炸** | 可能无限增长（LTP累积） | 自动归一化 |
| **权重消失** | 可能衰减到0（LTD累积） | 自动恢复 |
| **学习失衡** | 强化 > 抑制 → 爆炸 | 动态平衡 |
| **网络稳定性** | 不稳定（每次梦境偏移） | 长期稳定 |

### 3.2 种子生成算法详解

**日间残留（Day Residue）**：

```python
async def collect_day_residue(
    event_history: list[Any],
    memory_candidates: list[MemoryNode],
    workspace: Path | None,
) -> list[DreamSeed]:
    """收集日间残留种子（白天事件的余波）。"""
    seeds: list[DreamSeed] = []

    # 最近24小时的事件
    now = time.time()
    day_window = 24 * 3600
    day_events = [
        e for e in event_history
        if now - e.timestamp <= day_window
    ]

    # 按事件类型分类
    message_events = [e for e in day_events if e.event_type == EventType.MESSAGE]
    tool_events = [e for e in day_events if e.event_type == EventType.TOOL_CALL]

    # 消息事件 → 情感残留
    for event in message_events[-5:]:  # 最近5条消息
        text = event.content or ""

        # 计算情感唤醒度（使用LLM判断）
        affect_arousal = await estimate_emotional_arousal(text)

        # 计算重要性（是否有未决事项）
        importance = 0.5

        # 计算入梦可能性（情感唤醒度 × 未完成程度）
        dreamability = affect_arousal * 0.6

        seed = DreamSeed(
            seed_id=f"day_residue_{event.sequence}",
            seed_type=DreamSeedType.DAY_RESIDUE,
            title=f"白天的对话余波：{text[:30]}",
            summary=text[:100],
            affect_arousal=affect_arousal,
            importance=importance,
            dreamability=dreamability,
            score=affect_arousal * dreamability,
            tension_reason="白天对话留下的情感余波",
        )
        seeds.append(seed)

    return seeds
```

**未完成张力（Unfinished Tension）**：

```python
async def collect_unfinished_tension(
    memory_candidates: list[MemoryNode],
    workspace: Path | None,
) -> list[DreamSeed]:
    """收集未完成张力种子（未决任务、悬而未决的问题）。"""
    seeds: list[DreamSeed] = []

    # 查找TODO文件
    if workspace is None:
        return seeds

    todo_files = list(workspace.glob("todos/*.md"))
    for todo_file in todo_files[:5]:  # 最近5个TODO
        try:
            content = todo_file.read_text(encoding="utf-8")
        except Exception:
            continue

        # 检查TODO状态
        if "status: completed" in content.lower():
            continue  # 已完成的TODO不入梦

        # 提取标题
        title = todo_file.stem

        # 计算张力强度（未完成程度）
        # status: pending → 张力=0.8
        # status: in_progress → 张力=0.5
        tension_strength = 0.7 if "pending" in content.lower() else 0.5

        # 计算入梦可能性（未完成张力更容易入梦）
        dreamability = tension_strength * 0.8

        seed = DreamSeed(
            seed_id=f"unfinished_{todo_file.stem}",
            seed_type=DreamSeedType.UNFINISHED_TENSION,
            title=f"未完成的任务：{title}",
            summary=content[:100],
            affect_arousal=tension_strength * 0.5,
            importance=tension_strength,
            dreamability=dreamability,
            score=tension_strength * dreamability,
            tension_reason="任务未完成产生的心理张力",
            reference_file=str(todo_file.relative_to(workspace)),
        )
        seeds.append(seed)

    return seeds
```

**梦境滞后（Dream Lag）**：

```python
async def collect_dream_lag(
    memory_candidates: list[MemoryNode],
    workspace: Path | None,
) -> list[DreamSeed]:
    """收集梦境滞后种子（几天前的事件，延迟入梦）。"""
    seeds: list[DreamSeed] = []

    # 查找3-7天前的记忆节点
    now = time.time()
    lag_window_start = 3 * 24 * 3600  # 3天前
    lag_window_end = 7 * 24 * 3600    # 7天前

    for node in memory_candidates:
        if node.created_at < now - lag_window_end:
            continue  # 太久（超过7天）
        if node.created_at > now - lag_window_start:
            continue  # 太近（少于3天）

        # 计算滞后入梦可能性（梦境滞后效应）
        days_since = (now - node.created_at) / 86400
        lag_score = max(0.0, 1.0 - abs(days_since - 5.0) / 2.0)  # 5天前峰值

        seed = DreamSeed(
            seed_id=f"dream_lag_{node.node_id}",
            seed_type=DreamSeedType.DREAM_LAG,
            title=f"几天前的记忆：{node.title}",
            summary=node.title,
            affect_arousal=node.emotional_arousal,
            importance=node.importance,
            dreamability=lag_score * node.activation_strength,
            score=lag_score * node.importance,
            tension_reason="记忆延迟入梦（梦境滞后效应）",
        )
        seeds.append(seed)

    return seeds
```

```python
async def collect_self_theme(
    memory_candidates: list[MemoryNode],
) -> list[DreamSeed]:
    """收集自我主题种子（关于自我的核心记忆）。"""
    seeds: list[DreamSeed] = []

    for node in memory_candidates:

        # - 情感唤醒度高（emotional_arousal > 0.6）
        # - 重要性高（importance > 0.7）
        # - 激活强度高（activation_strength > 0.5）

            continue
        if node.importance < 0.7:
            continue
        if node.activation_strength < 0.5:

        seed = DreamSeed(
            seed_id=f"self_{node.node_id}",
            seed_type=DreamSeedType.SELF_THEME,
            title=f"自我主题记忆：{node.title}",
            summary=node.title,
            affect_arousal=node.emotional_arousal,
            importance=node.importance,
            dreamability=node.emotional_arousal * node.importance,
            score=node.emotional_arousal * node.importance * node.activation_strength,
            tension_reason="关于自我的核心记忆",
        )
        seeds.append(seed)

    return seeds[:3]  # 最多3个自我主题种子
```

### 3.3 种子选择机制

**海马体重复抑制（Hippocampal Repetition Suppression）**：

```python
# plugins/life_engine/dream/seeds.py::select_seed_candidates

def select_seed_candidates(
    candidates: list[DreamSeed],
    recent_seed_titles: set[str],
    repetition_decay: float = 0.3,
) -> list[DreamSeed]:
    """选择种子候选（应用海马体重复抑制）。"""
    selected: list[DreamSeed] = []

    # 按得分排序
    candidates.sort(key=lambda s: s.score, reverse=True)

    # 选择Top-K种子（应用重复抑制）
    for seed in candidates:
        # 检查是否最近已入梦（重复抑制）
        if seed.title in recent_seed_titles:
            # 最近入梦的种子得分降低
            seed.score *= (1.0 - repetition_decay)  # 降低30%

        # 如果得分仍然足够高，选择
        if seed.score > 0.3:  # 最低阈值
            selected.append(seed)

        # 最多选择5个种子
        if len(selected) >= 5:
            break

    return selected
```

**重复抑制的生物学原理**：

- **海马体重复抑制**：Hippocampus对重复刺激的反应逐渐减弱
- **意义**：避免梦境重复（相同主题连续入梦）
- **实现**：最近5次梦境的种子标题集合 → 重复种子得分降低30%

### 3.4 REM联想扩散

**REM游走详解**：

```python
# plugins/life_engine/dream/scheduler.py::run_rem

async def _run_rem(self, seed_node_ids: list[str]) -> REMReport:
    """REM阶段：从种子节点出发进行激活扩散。"""
    report = REMReport()

    if not self._memory:
        return report

    # 渐进参数（后半夜梦境更深）
    n = self._dreams_since_sleep_start  # 本晚已做梦境数量

    # 后半夜（n≥2）→ 扩散更远、衰减更慢、种子更多
    effective_depth = self._rem_max_depth + min(n, 3)  # 默认3 + n（最多6）
    effective_decay = min(self._rem_decay_factor + n * 0.05, 0.85)  # 默认0.6 + 0.05n
    effective_seeds = self._rem_seeds_per_round + n  # 默认5 + n
    effective_rounds = self._rem_walk_rounds + (1 if n >= 2 else 0)  # 默认2 + 1

    # 种子节点ID（最多effective_seeds个）
    actual_seed_ids = seed_node_ids[:effective_seeds]

    # 多轮游走
    for _ in range(effective_rounds):
        result = await self._memory.dream_walk(
            num_seeds=effective_seeds,
            seed_ids=actual_seed_ids or None,
            max_depth=effective_depth,
            decay_factor=effective_decay,
            learning_rate=self._rem_learning_rate,  # 默认0.05
        )

        report.nodes_activated += int(result.get("nodes_activated", 0) or 0)
        report.new_edges_created += int(result.get("new_edges_created", 0) or 0)
        report.walk_rounds += 1

        if not report.seed_node_ids:
            report.seed_node_ids = list(result.get("seed_ids") or [])

    # 弱边剪枝（清理无效连接）
    pruned = await self._memory.prune_weak_edges(
        threshold=self._rem_edge_prune_threshold,  # 默认0.08
    )

    if isinstance(pruned, dict):
        report.edges_pruned = int(pruned.get("pruned", 0) or 0)
    else:
        report.edges_pruned = int(pruned or 0)

    return report
```

**REM游走特性**：

| 维度 | 检索时激活扩散 | REM梦境游走 |
|-----|--------------|------------|
| **种子选择** | 查询驱动（Top-K） | 种子节点（入梦种子） |
| **扩散深度** | depth=2 | depth=3-6（后半夜更深） |
| **衰减系数** | decay=0.7 | decay=0.6-0.85（后半夜更慢） |
| **游走轮数** | 1轮 | 2-3轮（多轮联想） |
| **学习率** | rate=0.1 | rate=0.05（温和学习） |
| **剪枝** | 无 | threshold=0.08（弱边剪枝） |

**REM的系统级效果**：

- **新连接建立**：远距离联想 → 创造性连接（例如：预算焦虑 → 思考快与慢）
- **弱连接剪枝**：清理无效边 → 优化网络结构
- **记忆重组**：网络拓扑改变 → 记忆组织演化

```python
def wake_up(self):
    """觉醒过渡：恢复精力、释放压力。"""
    # 皮质醇晨峰（精力恢复）
    energy = self.modulators.get("energy")
    if energy:
        energy.value = min(energy.value + 0.25, 0.85)
        energy.baseline = 0.55  # 恢复正常基线

    # 睡眠后情绪稳态重置
    sociability = self.modulators.get("sociability")
    if sociability:
        sociability.baseline = 0.50
        sociability.value = max(sociability.value, 0.4)

    contentment = self.modulators.get("contentment")
    if contentment:
        contentment.value = min(contentment.value + 0.1, 0.7)
```

**觉醒过渡的意义**：
